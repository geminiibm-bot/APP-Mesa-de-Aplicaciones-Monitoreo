function doGet(e) {
  return HtmlService.createHtmlOutputFromFile('index')
    .setTitle("Mesa de Aplicaciones & Monitoreo")
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function enviarCorreoMonitoreo(destinatario, asunto, cuerpoTexto, cuerpoHtml) {
  GmailApp.sendEmail(destinatario, asunto, cuerpoTexto, {
    htmlBody: cuerpoHtml,
    name: "Mesa de Aplicaciones & Monitoreo"
  });
}
