function registrarAlerta(data) {
  try {
    var sheet = SpreadsheetApp.openById('1N8uoiQ0aF69YPgvaPccdTDTqjVoVvMfEXrU4pVN8b_w').getActiveSheet();
    var marcaTemporal = new Date(); // Fecha y hora automática

  }
}

function procesarDatosPSE(payload, horaInicio, horaFin) {
  try {
    // Decodificar el texto plano a un Array 2D nativo
    var jsonData = JSON.parse(payload);
    if (!jsonData || jsonData.length === 0) return "⚠️ Archivo vacío";
    
    // Usa el ID del archivo maestro de PSE
    var ss = SpreadsheetApp.openById('1N8uoiQ0aF69YPgvaPccdTDTqjVoVvMfEXrU4pVN8b_w');
    var hojaCausales = ss.getSheetByName("%Causales");
    if (!hojaCausales) return "❌ Error: No se encontró la hoja '%Causales'";
    
    // 1. Borrado seguro de todo el rango ocupado sin dañar la hoja
    hojaCausales.getDataRange().clearContent();
    SpreadsheetApp.flush(); // Forzar el borrado antes de pegar

    var maxCols = 0;
    for (var i = 0; i < jsonData.length; i++) {
      if (jsonData[i].length > maxCols) maxCols = jsonData[i].length;
    }
    
    // Cuadrar y limpiar el array para pegado perfecto
    var cleanData = [];
    for (var i = 0; i < jsonData.length; i++) {
      var row = [];
      for (var j = 0; j < maxCols; j++) {
        var cell = jsonData[i][j];
        
        // Asegurar que la columna "Cod. de autorización" (índice 5) o códigos similares
        // se peguen como texto en Google Sheets usando el apóstrofe inicial
        if (j === 5 && cell !== null && cell !== undefined && cell !== "") {
          cell = "'" + String(cell);
        } else if (typeof cell === 'string' && cell.match(/^0\d+$/)) {
          cell = "'" + cell;
        }
        
        row.push((cell === null || cell === undefined) ? "" : cell);
      }
      cleanData.push(row);
    }
    
    if (cleanData.length > 0 && maxCols > 0) {
      hojaCausales.getRange(1, 1, cleanData.length, maxCols).setValues(cleanData);
    }
    
    // Forzar que Google Sheets calcule las tablas dinámicas
    SpreadsheetApp.flush();
    Utilities.sleep(3000); // Pausa de 3 seg para asegurar que la tabla dinámica termine de refrescar
    
    // 2. Registrar horas, calcular porcentaje y registrar en Base
    var hojaDinamica = ss.getSheetByName("Tabla dinámica 2");
    var hojaBase = ss.getSheetByName("Base");
    if (hojaDinamica && hojaBase) {
      // Pegar los horarios en D16 y E16 antes de continuar
      hojaBase.getRange("D16").setValue(horaInicio);
      hojaBase.getRange("E16").setValue(horaFin);
      
      var numerador = hojaDinamica.getRange("A8159").getValue();
      var denominador = hojaDinamica.getRange("B671").getValue();
      var porcentaje = (denominador !== 0 && denominador !== "") ? (numerador / denominador) * 100 : 0;
      hojaBase.getRange("D18").setValue(porcentaje);
    }
    SpreadsheetApp.flush();
    
    // 3. Mover a Base de Datos Histórica
    var hojaDestino = ss.getSheetByName("DB CORTES PSE ABRIL");
    if (hojaDinamica && hojaDestino) {
      var filaValores = hojaDinamica.getRange("I2:Y2").getValues()[0]; // Extrae los 17 datos en una sola línea
      hojaDestino.appendRow(filaValores);
    }
    
    return "✅ Datos procesados, calculados y guardados en DB";
  } catch (e) {
    return "❌ Error interno: " + e.message;
  }
}

function obtenerDatosPorMapa(mapa) {
  var ss = SpreadsheetApp.openById('1N8uoiQ0aF69YPgvaPccdTDTqjVoVvMfEXrU4pVN8b_w');
  var sheet = ss.getSheetByName("Base"); // Ahora lee directamente del dashboard final
  var data = {};
  if (!sheet) return data;
  for (var key in mapa) {
    try {
      data[key] = sheet.getRange(mapa[key]).getDisplayValue() || "--";
    } catch(e) { data[key] = "--"; }
  }
  return data;
}

function obtenerHistorialGraficoPSE() {
  try {
    var ss = SpreadsheetApp.openById('1N8uoiQ0aF69YPgvaPccdTDTqjVoVvMfEXrU4pVN8b_w');
    // Se consume de la hoja del histórico de cortes.
    var sheet = ss.getSheetByName("DB CORTES PSE ABRIL");
    if (!sheet) return null;

    // Tomar los últimos 12 cortes para la gráfica (mínimo 10 solicitado).
    var lastRow = sheet.getLastRow();
    if (lastRow < 2) return null; // No hay datos además de la cabecera.
    var startRow = Math.max(2, lastRow - 11); 
    var numRows = lastRow - startRow + 1;
    
    // Se leen las primeras 15 columnas para asegurar que capturamos todo
    var rangeData = sheet.getRange(startRow, 1, numRows, 15).getValues(); 
    
    var labels = [];
    var aprobadas = [];
    var fallidas = [];
    var rechazadasSin15 = [];
    var causal15 = [];
    
    // ⚠️ CONFIGURACIÓN DE COLUMNAS: Ajusta el número si los datos no corresponden.
    // Recuerda que en Arrays el índice comienza en 0 (Col A = 0, B = 1, C = 2, D = 3...)
    var IDX_HORA = 0;             // Columna A (Hora del corte)
    var IDX_APROBADAS = 2;        // Columna C (Cantidad Aprobadas)
    var IDX_FALLIDAS = 6;         // Columna G (% TX Rango Fallidas)
    var IDX_CAUSAL_15 = 8;        // Columna I (Cantidad Causal 15)
    var IDX_RECH_SIN_15 = 5;      // Columna F (Sin Causal 00015)

    for (var i = 0; i < rangeData.length; i++) {
      var row = rangeData[i];
      if (row[IDX_HORA] !== "" && row[IDX_HORA] != null) {
        var lbl = row[IDX_HORA] instanceof Date ? Utilities.formatDate(row[IDX_HORA], Session.getScriptTimeZone(), "HH:mm") : String(row[IDX_HORA]);
        labels.push(lbl);
        aprobadas.push(Math.round(Number(row[IDX_APROBADAS])) || 0);
        fallidas.push(Math.round(Number(row[IDX_FALLIDAS])) || 0);
        causal15.push(Math.round(Number(row[IDX_CAUSAL_15])) || 0);
        rechazadasSin15.push(Math.round(Number(row[IDX_RECH_SIN_15])) || 0);
      }
    }

    if (labels.length > 0) {
      return { labels: labels, aprobadas: aprobadas, fallidas: fallidas, rechazadasSin15: rechazadasSin15, causal15: causal15 };
    }
    return null;
  } catch (e) {
    return null;
  }
}
